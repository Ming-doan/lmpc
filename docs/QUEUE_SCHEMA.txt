# ---------- Task queue ----------
queue:benchmarks                   LIST           run_id values; BRPOPLPUSH into processing
queue:benchmarks:processing        LIST           reliable-queue pattern; worker removes on success
queue:benchmarks:dead              LIST           moved here after max retries

# ---------- Run state (hot) ----------
run:{run_id}:state                 HASH           { status, current_step, started_at, worker_id,
                                                    inference_container_id, last_heartbeat }
run:{run_id}:cancel                STRING "1"     set by admin to request cancellation; polled by worker
run:{run_id}:lock                  STRING         NX + EX 60s, refreshed; prevents double-pickup

# ---------- Step event log (for dashboard replay) ----------
run:{run_id}:steps                 STREAM         XADD entries: { step_name, step_status,
                                                                   progress_pct, message, details }
                                                  Consumer: SSE handler does XREAD BLOCK.

# ---------- Metrics (time series) ----------
run:{run_id}:metrics:series        STREAM         XADD every ~1s: { t_offset_ms, concurrency,
                                                                     latency_ms, throughput_tps,
                                                                     ttft_ms, tps, cpu_pct,
                                                                     gpu_pct, ram_mb, vram_mb }
run:{run_id}:metrics:latest        HASH           last sample; O(1) read for dashboard "current" tiles
run:{run_id}:metrics:running_avg   HASH           { sum_latency, n_latency, sum_tps, n_tps, ... }
                                                  running accumulators; used to compute avg_* on
                                                  finalize without scanning the full stream

# ---------- Pub/Sub fanout (SSE) ----------
events:run:{run_id}                PUBSUB         messages: { type: 'step'|'metric'|'status', payload }
                                                  backend subscribes per SSE connection and forwards

# ---------- Single-run coordination (per worker host) ----------
worker:{worker_id}:active_run      STRING         run_id currently executing on this worker; SET NX
                                                  worker refuses to start a second run on same host
                                                  (replaces global host:active_run — supports multi-worker)

# ---------- Worker liveness ----------
# Workers do NOT connect to Redis directly. They heartbeat via POST /internal/workers/{id}/heartbeat
# to the API, which updates workers.last_heartbeat in Postgres and refreshes this key.
worker:{worker_id}:heartbeat       STRING         TTL 15s, refreshed by API on each worker heartbeat;
                                                  reaper looks for dead workers holding processing entries
worker:{worker_id}:info            HASH           { gpu_model, vram_mb, cpu, ram_mb, registered_at }
                                                  written by API on /internal/workers/register

# ---------- Model metadata cache (optional; mirrors models_cache) ----------
model:{source}:{model_id}          HASH + TTL     short-TTL cache in front of Postgres lookups
